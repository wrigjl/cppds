Exploration 8.4.1. Nodes and References BinaryTree.

(a) C++ Implementation. 

#include <iostream>
#include <cstdlib>
using namespace std;

class BinaryTree {
private:
    char key;
    BinaryTree *leftChild;
    BinaryTree *rightChild;

public:
    BinaryTree(char rootObj){
        this->key = rootObj;
        this->leftChild = NULL;
        this->rightChild = NULL;
    }
};
