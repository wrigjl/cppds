Exploration 1.13.3. Implementation of UnaryGate.

(a) C++ implementation. 
class UnaryGate : public LogicGate {
public:
        UnaryGate(string n) : LogicGate(n) {
                pinTaken = false;
        }
        bool getPin() {
                if (pinTaken==false) {
                        cout << "Enter Pin input for gate " << getLabel() << ": ";
                        cin >> pin;
                        pinTaken = true;
                }
                return pin;
        }
protected:
        bool pin;
        bool pinTaken;
};
