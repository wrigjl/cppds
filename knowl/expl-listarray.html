(a) C++ Implementation. 
//showcases an iteration through an array in C++
#include <iostream>
using namespace std;

int main(){
    int myarray[] = {2,4,6,8};
    for (int i=0; i<4; i++){
        cout << myarray[i] << endl;
    }
    return 0;
}
