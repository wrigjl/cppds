//Example of summing a vector by using recursion.

#include <iostream>
#include <vector>
using namespace std;

int vectsum(vector<int> numVect){
    if (numVect.size() == 0)
        return 0;

    if (numVect.size() == 1)
        return numVect[0];

    // make a copy of the vector without the first element
    vector<int> slice(numVect.begin() + 1, numVect.end());

    //function makes a recursive call to itself.
    return numVect[0] + vectsum(slice);
}

int main() {
    vector<int> numVect = {1, 3, 5, 7, 9};
    cout << vectsum(numVect) << endl;
    return 0;
}
