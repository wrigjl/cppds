class BinaryGate : public LogicGate {
public:
        BinaryGate(string n) : LogicGate(n) { // When we create an instance of
        // BinaryGate, data inherited from LogicGate are initialized with n;
                pinATaken = false;
                pinBTaken = false;
        }

        bool getPinA() {
                if (pinATaken==false) {
                        cout << "Enter Pin input for gate " << getLabel() << " : ";
                        cin >> pinA;
                        pinATaken = true;
                }
                return pinA;
        }

        bool getPinB() {
                if (pinBTaken==false ) {
                        cout << "Enter Pin input for gate " << getLabel() << " : ";
                        cin >> pinB;
                        pinBTaken = true;
                }
                return pinB;
        }

protected:
        bool pinA;
        bool pinATaken;
        bool pinB;
        bool pinBTaken;
};
