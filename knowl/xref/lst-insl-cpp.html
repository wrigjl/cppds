void insertLeft(char newNode){
    if (this->leftChild == nullptr){
        this->leftChild = new BinaryTree(newNode);
    }
    else {
        BinaryTree *t = new BinaryTree(newNode);
        t->leftChild = this->leftChild;
        this->leftChild = t;
    }
}
