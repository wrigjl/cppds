void insertRight(char newNode){
    if (this->rightChild == NULL){
        this->rightChild = new BinaryTree(newNode);
    }
    else {
        BinaryTree *t = new BinaryTree(newNode);
        t->rightChild = this->rightChild;
        this->rightChild = t;
    }
}
