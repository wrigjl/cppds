Exploration 1.13.4. Implementation of AndGate.

(a) C++ implementation. 
class AndGate : public BinaryGate {
public:
        AndGate(string n) : BinaryGate(n) {};

        virtual bool performGateLogic() {
                bool a = getPinA();
                bool b = getPinB();
                if (a == 1 && b == 1) {
                        return true;
                }
                else {
                        return false;
                }
        }
};
